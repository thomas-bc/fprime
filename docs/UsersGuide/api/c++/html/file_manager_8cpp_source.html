261  ) const
262  {
263  const char evalStr[] = "eval '%s' 1>>%s 2>&1\n";
264  const U32 bufferSize = sizeof(evalStr) - 4 + 2 * FW_CMD_STRING_MAX_SIZE;
265  char buffer[bufferSize];
266 
267  NATIVE_INT_TYPE bytesCopied = snprintf(
268  buffer, sizeof(buffer), evalStr,
269  command.toChar(),
270  logFileName.toChar()
271  );
272  FW_ASSERT(static_cast<NATIVE_UINT_TYPE>(bytesCopied) < sizeof(buffer));
273 
274  const int status = system(buffer);
